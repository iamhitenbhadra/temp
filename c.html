1. Program to find nPr of a number using functions

#include <stdio.h> 
 
int factorial(int n) { 
    int fact = 1, i; 
    for (i = 1; i <= n; i++) { 
        fact = fact * i; 
    } 
    return fact; 
} 
 
int main() { 
    int n, r, npr; 
 
    printf("Enter n and r: "); 
    scanf("%d %d", &n, &r); 
 
    if (n >= r) { 
        npr = factorial(n) / factorial(n - r); 
        printf("nPr = %d\n", npr); 
    } else { 
        printf("Invalid input. n should be greater than or equal to r.\n"); 
    } 
 
    return 0; 
}

4. Function to find GCD of two numbers and use it to find GCD of 4 numbers

#include <stdio.h> 
 
int gcd(int x, int y) { 
    while (x != y) { 
        if (x > y) 
            x = x - y; 
        else 
            y = y - x; 
    } 
    return x; 
} 
 
int main() { 
    int a, b, c, d, result; 
 
    printf("Enter four numbers: "); 
    scanf("%d %d %d %d", &a, &b, &c, &d); 
 
    result = gcd(a, b); 
    result = gcd(result, c); 
    result = gcd(result, d); 
 
    printf("GCD of four numbers = %d\n", result); 
 
    return 0; 
}

5. Program to find factorial of a given number using recursive function

#include <stdio.h> 
 
int factorial(int n) { 
    if (n == 0 || n == 1) 
        return 1; 
    else 
        return n * factorial(n - 1); 
} 
 
int main() { 
    int n; 
 
    printf("Enter a number: "); 
    scanf("%d", &n); 
 
    printf("Factorial of %d = %d\n", n, factorial(n)); 
 
    return 0; 
}

7. Program to find GCD of two numbers using recursive function

#include <stdio.h> 
 
int gcd(int a, int b) { 
    if (a == b) 
        return a; 
    else if (a > b) 
        return gcd(a - b, b); 
    else 
        return gcd(a, b - a); 
} 
 
int main() { 
    int a, b; 
 
    printf("Enter two numbers: "); 
    scanf("%d %d", &a, &b); 
 
    printf("GCD = %d\n", gcd(a, b)); 
 
    return 0; 
}

13. Program to sort an array in ascending order using bubble sort algorithm

#include <stdio.h> 
 
int main() { 
    int arr[50], n, i, j, temp; 
 
    printf("Enter size of array: "); 
    scanf("%d", &n); 
 
    printf("Enter array elements:\n"); 
    for (i = 0; i < n; i++) { 
        scanf("%d", &arr[i]); 
    } 
 
    for (i = 0; i < n - 1; i++) { 
        for (j = 0; j < n - 1 - i; j++) { 
            if (arr[j] > arr[j + 1]) { 
                temp = arr[j]; 
                arr[j] = arr[j + 1]; 
                arr[j + 1] = temp; 
            } 
        } 
    } 
 
    printf("Sorted array in ascending order:\n"); 
    for (i = 0; i < n; i++) { 
        printf("%d ", arr[i]); 
    } 
    printf("\n"); 
 
    return 0; 
}

19. Program to count number of alphabets, numbers and other characters in a string

#include <stdio.h> 
#include <ctype.h> 
 
int main() { 
    char str[100]; 
    int i, alphabets = 0, digits = 0, others = 0; 
 
    printf("Enter a string: "); 
    fgets(str, sizeof(str), stdin); 
 
    for (i = 0; str[i] != '\0' && str[i] != '\n'; i++) { 
        if (isalpha(str[i])) 
            alphabets++; 
        else if (isdigit(str[i])) 
            digits++; 
        else 
            others++; 
    } 
 
    printf("Number of alphabets = %d\n", alphabets); 
    printf("Number of digits = %d\n", digits); 
    printf("Number of other characters = %d\n", others); 
 
    return 0; 
}

20. Program to exchange two numbers using functions

#include <stdio.h> 
 
int a, b; 
 
void exchange() { 
    int temp; 
    temp = a; 
    a = b; 
    b = temp; 
} 
 
int main() { 
    printf("Enter two numbers: "); 
    scanf("%d %d", &a, &b); 
 
    printf("Before exchange: a = %d, b = %d\n", a, b); 
    exchange(); 
    printf("After exchange: a = %d, b = %d\n", a, b); 
 
    return 0; 
}

22. Program using pointers to find sum of a given 1-D array

#include <stdio.h> 
 
int main() { 
    int arr[50], n, i, sum = 0; 
    int *p; 
 
    printf("Enter size of array: "); 
    scanf("%d", &n); 
 
    printf("Enter array elements:\n"); 
    for (i = 0; i < n; i++) { 
        scanf("%d", &arr[i]); 
    } 
 
    p = arr; 
 
    for (i = 0; i < n; i++) { 
        sum = sum + *p; 
        p++; 
    } 
 
    printf("Sum of array elements = %d\n", sum); 
 
    return 0; 
}
