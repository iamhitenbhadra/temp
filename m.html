1a.
Interface Push button to control LED and Buzzer with ESP32
CPP
void setup() {
  pinMode(14,OUTPUT); // Buzzer
  pinMode(2, OUTPUT); // LED
  pinMode(35,INPUT);  // Push Button
}
void loop() {
  if(digitalRead(35)==HIGH) {
    digitalWrite(14,HIGH);
    digitalWrite(2, HIGH);
  } else {
    digitalWrite(14,LOW);
    digitalWrite(2, LOW);
  }
}

2a.
Interface Touch Sensor with an ESP32 to control the onboard LED and displays touch event notifications on an LCD screen.
CPP
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
LiquidCrystal_I2C lcd(0x27,16,2); 
#define tsPin 18 
int ledPin = 2; 

void setup() {
  lcd.init();
  lcd.backlight();
  Serial.begin(9600); 
  pinMode(ledPin, OUTPUT); 
  pinMode(tsPin, INPUT); 
}
void loop() {
  int tsValue = digitalRead(tsPin);
  delay(100);
  if (tsValue==HIGH) {
    digitalWrite(ledPin, HIGH); 
    lcd.setCursor(0,0);
    lcd.print("Touched    ");
  } else {
    digitalWrite(ledPin, LOW); 
    lcd.setCursor(0,0);
    lcd.print("NoT_Touched");
  }
}

3a.
Interface an LDR sensor with an ESP32 to control a 240V AC light such that the light turns ON automatically when the ambient light intensity is less than 500 and turns OFF when the intensity exceeds 500.
CPP
void setup() {
  Serial.begin(9600);
  pinMode(36, INPUT);  // LDR
  pinMode(13, OUTPUT); // Relay controlling 240V AC light
}
void loop() {
  int ldrValue = analogRead(36);
  if(ldrValue < 500) {
    digitalWrite(13, HIGH); 
  } else {
    digitalWrite(13, LOW); 
  }
  delay(100);
}

4a.
Interface a 16x2 LCD with ESP32 and display a left scrolling message continuously across the LCD screen.
CPP
#include <LiquidCrystal_I2C.h>
#include <Wire.h>
LiquidCrystal_I2C lcd(0x27, 16, 2);

void setup() {
  lcd.init();
  lcd.backlight();
  lcd.setCursor(0,0);
  lcd.print("Welcome to BNMIT Bangalore");
  delay(1000);
}
void loop() {
  for (int positionCounter = 0; positionCounter < 13; positionCounter++) {
    lcd.scrollDisplayLeft();
    delay(150);
  }
  delay(1000);
}

      digitalWrite(14, LOW);
    }
  }
}

8a.
Interface DHT11 Sensor with ESP32 and display Humidity and Temperature value on LCD display.
CPP
#include "DHT.h"
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#define DHTPIN 32
LiquidCrystal_I2C lcd(0x27,16,2); 
DHT dht(DHTPIN, DHT11);

void setup() {
  lcd.init();
  lcd.backlight();
  dht.begin();
}
void loop() {
  float humi = dht.readHumidity();
  float temp = dht.readTemperature();
  
  lcd.setCursor(0,0);
  lcd.print("Temp: ");
  lcd.print(temp);
  lcd.print("C");
  
  lcd.setCursor(0,1);
  lcd.print("Humidity: ");
  lcd.print(humi);
  
  delay(1000);
}

Interface ESP32 with Wi Fi through Blynk app to control the LED.
CPP
#define BLYNK_TEMPLATE_ID "TMPLW7uIJFDM"
#define BLYNK_DEVICE_NAME "LED"
#define BLYNK_AUTH_TOKEN "aCixJKNOQK43PbB4uVy29yZgq6ZFqOSj"
#define LED 2
#define BLYNK_PRINT Serial
#include <WiFi.h>
#include <BlynkSimpleEsp32.h>

char auth[] = BLYNK_AUTH_TOKEN;
char ssid[] = "abcd";
char pass[] = "ZXCV@098";
BlynkTimer timer;

BLYNK_WRITE(V1) {
  int value = param.asInt();
  Blynk.virtualWrite(V1, value);
  if(value == 1){
    digitalWrite(LED, HIGH);
  } else if(value == 0){
    digitalWrite(LED, LOW);
  }
}
void setup() {
  pinMode(LED,OUTPUT);
  Serial.begin(115200);
  Blynk.begin(auth, ssid, pass);
}
void loop() {
  Blynk.run();
  timer.run();
}
